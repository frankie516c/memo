리액트의 기본 함수형을 태그형식으로 만들면 JSX로 리턴해주면 함수 첫번재는 무조건 대문자로 사용하고 요소로 값을 전달할 수 있고 저렇게 `function Main({arr, list})` 처럼 실행 가능
App.js
```
import logo from './logo.svg';
import './App.css';

const Page1 = () => <h1>상단 화면</h1>;
const Page2 = () => <h1>하단 화면</h1>;

function Main({arr, list}) {
  console.log("Main()", arr, list);
  return (
   <main>
    {
      arr?.map((v, i) => <section key={i}>{v}</section>)
    }
   </main>
  )
}

function App() {

  const arr = ["내용", "설명", "보조 내용"];

  return (
    <>
    <Page1 />

    <Main list={arr} />

    <Page2 />
    </>
  );
}

export default App;
```
