App.jsx 에 다음과 같이 코드를 적용하면 됨
```
const Page1 = () => <h1>상단 화면</h1>
const Page2 = () => <h1>하단 화면</h1>

function Main () {
  const arr = ["1번", "2번", "3번"]
  return ( 
    <ul>
      {
        arr.map((v,i) => <li key={i}>{v}</li>)
      }
    </ul>
  )
}

function App() {
  return (
    <> 
      <Page1 />
      
      <Main />

      <Page2 />
    </>
  )
}

export default App
```
