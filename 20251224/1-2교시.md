배열을 제대로 선택하려면 아래와 같이 입력해야 함<br />
document.getElementsByTagName("section")[0];<br />

`document.getElementsByTagName("section")[0].innerHTML = '<div class="bg2"></div>';`<br />

문서처럼 만들어지는 글을 ㅇ릭는건 브라우저고 브라우저가 이 **순서**를 넣어야 하기 때문에 필요함<br />

}

console.log("스터디01")
function load() {
var html = "";
for(let i = 0; i < 25; i++ ) {
    let css = "1";
    if(i % 2 === 0) css = "2"
        html += `<div class="bg${ css }"></div>`;
}
document.getElementsByTagName("section")[0].innerHTML = html;
}

function clean() {
    document.getElementsByTagName("section")[0].innerHTML = "";
}
var state = false; 
function btnToggle() {
    if(!state) load();
    else clean();
    state = !state;
}
